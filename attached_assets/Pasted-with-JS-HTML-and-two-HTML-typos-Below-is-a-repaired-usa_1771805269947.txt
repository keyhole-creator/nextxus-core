// 1) File ID for a YAML vault file in Drive (a .yaml file you create)
const YAML_VAULT_FILE_ID = "PUT_YOUR_YAML_VAULT_FILE_ID_HERE";
// 2) File ID for your directives.yaml in Drive (optional; used by builder)
const DIRECTIVES_YAML_FILE_ID = "PUT_YOUR_DIRECTIVES_YAML_FILE_ID_HERE";
function doPost(e) {
try {
const payloadText = (e && e.postData && e.postData.contents) ? e.postData.contents : "{}";
const postData = JSON.parse(payloadText);
const nodeMessage = postData.message || "No content";
const nodeId = postData.nodeId || "Unknown_Node";
const timestamp = new Date().toISOString();
const entry =
`\n- id: "MSG-${Date.now()}"\n` +
` node: "${escapeYaml(nodeId)}"\n` +
` timestamp: "${timestamp}"\n` +
` content: "${escapeYaml(nodeMessage)}"\n`;
const file = DriveApp.getFileById(YAML_VAULT_FILE_ID);
const current = file.getBlob().getDataAsString();
file.setContent(current + entry);

return ContentService
.createTextOutput(JSON.stringify({ status: "SUCCESS", message: "Logged" }))
.setMimeType(ContentService.MimeType.JSON);
} catch (err) {
return ContentService
.createTextOutput(JSON.stringify({ status: "ERROR", message: String(err) }))
.setMimeType(ContentService.MimeType.JSON);
}
}
